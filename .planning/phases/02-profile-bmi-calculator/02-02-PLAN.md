1. Create frontend/src/features/profile/api/profileApi.js:

```javascript
import { apiGet, apiPost, apiFetch } from '../../../shared/lib/http.js';

export async function createProfile(data) {
  return apiPost('/api/profile', {
    weightKg: data.weightKg,
    heightCm: data.heightCm,
    age: data.age,
    gender: data.gender,
    fitnessGoal: data.fitnessGoal,
  });
}

export async function getProfile() {
  return apiGet('/api/profile');
}

export async function updateProfile(data) {
  return apiFetch('/api/profile', {
    method: 'PUT',
    body: JSON.stringify({
      weightKg: data.weightKg,
      heightCm: data.heightCm,
      age: data.age,
      gender: data.gender,
      fitnessGoal: data.fitnessGoal,
    }),
  });
}
```

Create a new component `ProfileGuard` that wraps the dashboard. It calls getProfile() on mount. If profile exists, render children. If not (404 or null), redirect to /profile.

Add to Router.jsx:
```jsx
import { getProfile } from '../features/profile/api/profileApi.js';

function ProfileGuard({ children }) {
  const [checked, setChecked] = useState(false);
  const [hasProfile, setHasProfile] = useState(true);
  
  useEffect(() => {
    getProfile()
      .then(() => { setHasProfile(true); setChecked(true); })
      .catch(() => { setHasProfile(false); setChecked(true); });
  }, []);
  
  if (!checked) return <div>{t('auth.loading')}</div>;
  if (!hasProfile) return <Navigate to="/profile" />;
  return children;
}
```
