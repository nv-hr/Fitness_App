       - Import userRepository from '../repositories/user.repository.js'
       - Import bcrypt from 'bcrypt'
       - Import jwt from 'jsonwebtoken'
       - Import { ValidationError, AuthenticationError } from '../utils/errors.js'
       - Export async function register({ email, password, pdpConsent }):
         a. Validate pdpConsent is true — throw ValidationError('PDP consent is required') if false (per D-02)
         b. Check if user exists via userRepository.findByEmail(email) — throw ValidationError('Email already registered') if found
         c. Hash password with bcrypt.hash(password, 10) (per D-05, 10 salt rounds)
         d. Create user via userRepository.create({ email, passwordHash: hashedPassword, pdpConsent: true })
         e. Generate JWT via generateToken(user)
         f. Return { user: { id, email, pdp_consent }, token }
       - Export async function login({ email, password }):
         a. Find user via userRepository.findByEmail(email) — throw AuthenticationError('Invalid email or password') if not found
         b. Compare password via bcrypt.compare(password, user.password_hash) — throw AuthenticationError('Invalid email or password') if false
         c. Generate JWT via generateToken(user)
         d. Return { user: { id, email, pdp_consent }, token }
       - Export async function handleGoogleOAuth({ googleId, email, displayName }):
         a. Find existing user via userRepository.findByGoogleId(googleId)
         b. If found: generate JWT, return { user, token }
         c. If not found: create user via userRepository.create({ email, passwordHash: null, googleId, pdpConsent: true }) — OAuth implies consent
         d. Generate JWT, return { user, token }
       - Export function generateToken(user): jwt.sign({ userId: user.id, email: user.email }, process.env.JWT_SECRET, { algorithm: 'HS256', expiresIn: '7d' }) (per CONTEXT.md: HS256, 7d expiry)

    2. backend/src/middlewares/auth.middleware.js — JWT verification middleware (per D-01):
       - Import jwt from 'jsonwebtoken'
       - Import { AuthenticationError } from '../utils/errors.js'
       - Export function authenticateToken(req, res, next):
         a. Read token from req.cookies.token (httpOnly cookie — per D-01, NOT from Authorization header)
         b. If no token: throw AuthenticationError('Authentication required')
         c. Verify token: jwt.verify(token, process.env.JWT_SECRET, { algorithms: ['HS256'] })
         d. Attach decoded user to req.user (req.user = { userId, email })
         e. Call next()
         f. Catch jwt.JsonWebTokenError: throw AuthenticationError('Invalid or expired token')
  </action>

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → POST /api/auth/register | Untrusted input: email, password, pdpConsent boolean |
| Client → POST /api/auth/login | Untrusted input: email, password — brute force target |
| Client → GET /api/auth/google/callback | OAuth callback with Google auth code — code injection risk |
| Backend → JWT cookie | Token sent to client — must be httpOnly to prevent XSS theft |
| Backend → bcrypt.compare | Timing attack surface on password comparison |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-06 | Spoofing | POST /api/auth/login | mitigate | Rate limiter: max 10 attempts per 15 minutes on /api/auth/login. Same error message for "user not found" and "wrong password" prevents email enumeration |
| T-01-07 | Tampering | JWT token in cookie | mitigate | httpOnly flag prevents JavaScript access. SameSite='lax' prevents CSRF from cross-site requests. HS256 signature verified on every request |
| T-01-08 | Information Disclosure | GET /api/auth/me | mitigate | Protected by authenticateToken middleware — requires valid JWT. Returns only { id, email, pdp_consent } — never password_hash or google_id |
| T-01-09 | Spoofing | Google OAuth callback | mitigate | Passport validates Google auth code server-side. state parameter prevents CSRF on OAuth flow (Passport handles automatically). session: false prevents Passport session storage |
| T-01-10 | Denial of Service | POST /api/auth/register | mitigate | Rate limiter: max 10 registrations per 15 minutes. express-validator rejects malformed emails before hitting database |
| T-01-11 | Repudiation | PDP consent capture | mitigate | pdp_consent boolean and pdp_consent_date timestamp stored in users table. Registration rejects if pdpConsent !== true. Audit trail via pdp_consent_date |
| T-01-12 | Elevation of Privilege | authenticateToken middleware | mitigate | jwt.verify with HS256 algorithm and algorithms: ['HS256'] restriction prevents algorithm confusion attacks. JWT_SECRET from environment variable, not hardcoded |
</threat_model>
